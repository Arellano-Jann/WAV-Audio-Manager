{\bfseries{How To Use This Application}}

\DoxyHorRuler{0}



\begin{DoxyItemize}
\item Build Application
\item Enter “./wavman” in Your Terminal to Run the Application
\item Enter the Name of the .wav File You Wish to Edit
\begin{DoxyItemize}
\item OR Type “q” to Exit the Program
\end{DoxyItemize}
\item The Application will Display the File’s Metadata, and Ask You to Choose a Processing Method to Apply to Your .wav File
\item Enter a Number Based on Which Processor You Would Like to Use
\begin{DoxyItemize}
\item (1) Normalization\+:
\begin{DoxyItemize}
\item This Processing Method Will Maximize the Amplitude of Your File
\item This Processing Method Will Automatically Apply to Your File and Return You to the “\+Choose a Processing Method” Page
\end{DoxyItemize}
\item (2) Echo\+:
\begin{DoxyItemize}
\item This Processing Method Will Create an Echo Effect from Your File
\item This Processing Method Will Ask You to Enter a Percentage to Scale the Echo’s Volume, then it Will Ask You to Enter a Delay Value by Number of Samples, and Return You to the “\+Choose a Processing Method” Page
\end{DoxyItemize}
\item (3) Gain
\begin{DoxyItemize}
\item This Processing Method Will Raise or Lower the Overall Amplitude of Your File
\item This Processing Method Will Ask You to Enter a Percentage to Scale the Overall Amplitude, and Return You to the “\+Choose a Processing Method” Page
\end{DoxyItemize}
\item (4) Low-\/\+Pass Filter
\begin{DoxyItemize}
\item This Processing Method Does Not Work Properly
\item Avoid Using This Processing Method
\end{DoxyItemize}
\item (5) Compression
\begin{DoxyItemize}
\item This Processing Method Will Scale the Volume of Your .wav File by a Ratio Based on a Maximum Threshold
\item This Processing Method Will Ask You to Enter a Ratio that Will Scale Aspects of Your File that Exceed the Maximum Threshold, then it Will Ask You to Set a Maximum Threshold 
\end{DoxyItemize}
\end{DoxyItemize}
\end{DoxyItemize}
